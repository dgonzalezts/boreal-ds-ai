# Colibri Icon Library

A comprehensive icon system for the Telesign Design System, built with Lit components and optimized SVG loading.

## Installation

```bash
npm install @telesign/colibri-icons
```

## Basic Usage

### Register the Component

```typescript
import { ColIcon } from '@telesign/colibri-icons';

// Register the custom element
customElements.define('col-icon', ColIcon);
```

### Use in HTML

```html
<col-icon name="message" color="#000000" size="24px"></col-icon>
```

### Use in JavaScript/TypeScript

```typescript
import { html } from 'lit';

html` <col-icon name="lock" color="currentColor" size="20px"></col-icon> `;
```

## Advanced Usage

### Dynamic Color Inheritance

The icon component can inherit colors from its parent:

```html
<div style="color: blue">
  <col-icon name="user" color="currentColor"></col-icon>
</div>
```

### Size Variations

Supports both pixel and relative units:

```html
<!-- Pixel sizes -->
<col-icon name="search" size="24px"></col-icon>

<!-- Number (automatically converted to pixels) -->
<col-icon name="menu" size="32"></col-icon>

<!-- Relative units -->
<col-icon name="settings" size="2rem"></col-icon>
```

### Icon Caching

Icons are automatically cached for better performance:

```typescript
// First load - fetches from file system
<col-icon name="user"></col-icon>

// Subsequent loads - uses cached version
<col-icon name="user"></col-icon>
```

## Component API

### Properties

| Property | Type             | Default          | Description          |
| -------- | ---------------- | ---------------- | -------------------- |
| `name`   | `string`         | `'lock'`         | Icon name to display |
| `color`  | `string`         | `'currentColor'` | Icon color           |
| `size`   | `string\|number` | `'24px'`         | Icon size            |

### CSS Custom Properties

The component can be styled using CSS:

```css
col-icon {
  display: inline-block;
  line-height: 0;
}

.icon-container {
  width: 100%;
  height: 100%;
}
```

## Adding Icons

### 1. Prepare SVG Files

- Export SVG from design tools (e.g., Figma)
- Ensure proper naming convention
- Optimize SVG content

### 2. Add to Project

```bash
# Add SVG file to icons directory
src/
  icons/
    my-new-icon.svg
```

### 3. SVG Requirements

- Monochromatic design
- Clean, optimized markup
- Proper viewBox attribute
- No hardcoded colors (unless using `currentColor`)

Example of a proper SVG:

```svg
<svg viewBox="0 0 24 24" fill="none" xmlns="http://www.w3.org/2000/svg">
  <path d="M12 2L2 7l10 5 10-5-10-5z" fill="currentColor"/>
</svg>
```

## Best Practices

### 1. Icon Naming

```typescript
// Good
name = 'user-profile';
name = 'shopping-cart';

// Avoid
name = 'icon1';
name = 'newicon';
```

### 2. Color Management

```typescript
// Good - Inherits from parent
<col-icon color="currentColor"></col-icon>

// Good - Specific color
<col-icon color="#FF0000"></col-icon>

// Avoid - Hardcoded colors in SVG
<svg fill="#000000">...</svg>
```

### 3. Size Handling

```typescript
// Good - Clear sizing
<col-icon size="24px"></col-icon>
<col-icon size="2rem"></col-icon>

// Avoid - Unclear units
<col-icon size="24"></col-icon>
```

## Troubleshooting

### Common Issues

1. Icon Not Loading

```typescript
// Check if icon exists in correct location
// Ensure proper name matching
<col-icon name="exact-file-name"></col-icon>
```

2. Color Not Applying

```typescript
// Ensure SVG doesn't have hardcoded colors
// Use currentColor in SVG
<col-icon color="blue"></col-icon>
```

3. Size Issues

```typescript
// Be explicit with units
<col-icon size="24px"></col-icon>  // Good
<col-icon size="24"></col-icon>    // May cause issues
```

## Performance Considerations

1. Icon Caching

- Icons are cached automatically
- Cache is shared across instances
- Cache is keyed by icon name and color

2. SVG Optimization

- SVGs are minified during build
- Unnecessary attributes are removed
- Inline styles are optimized

3. Load Time

- Icons are loaded on demand
- Failed loads are handled gracefully
- Cache prevents unnecessary reloads

import { html, LitElement } from 'lit';
import { property, state } from 'lit/decorators.js';
import { unsafeHTML } from 'lit/directives/unsafe-html.js';

export class ColIcon extends LitElement {
  static readonly tagName = 'icon';
  @property({ type: String })
  color: string = 'currentColor';
  @property({ type: String })
  name: string = 'lock';
  @property({ type: String })
  size: string | number = '24px';

  @state()
  private svgContent: string = '';
  private static iconCache = new Map<string, string>();

  private cleanSvgContent(content: string): string {
    try {
      let cleaned = content
        .replace(/export default /g, '')
        .replace(/^["']|["']$/g, '')
        .replace(/^data:image\/svg\+xml,/, '')
        .trim();

      if (!cleaned.includes('<svg')) {
        return '';
      }

      if (!cleaned.includes('width=') && !cleaned.includes('height=')) {
        cleaned = cleaned.replace('<svg', '<svg width="100%" height="100%"');
      }

      // Set the icon color
      if (!cleaned.match('fill') && !cleaned.match('stroke')) {
        cleaned = cleaned.replace('<svg', `<svg fill="${this.color}"`);
      } else {
        cleaned = cleaned.replace(/fill="([^none][\w]+)"/g, `fill="${this.color}"`);
        cleaned = cleaned.replace(/stroke="([^none][\w]+)"/g, `stroke="${this.color}"`);
        cleaned = cleaned.replace(/currentColor/g, this.color);
      }
      return cleaned;
    } catch (error) {
      console.error('Error cleaning SVG:', error);
      return '';
    }
  }

  async loadIcon(name: string): Promise<string> {
    const cacheKey = `${name}-${this.color}`;
    try {
      if (ColIcon.iconCache.has(cacheKey)) {
        return ColIcon.iconCache.get(cacheKey) || '';
      }

      const module = await import(`../icons/${name}.svg`);

      if (!module.default) {
        throw new Error('No SVG content loaded');
      }

      const cleanedSvg = this.cleanSvgContent(module.default);

      if (!cleanedSvg) {
        throw new Error('Invalid SVG content');
      }

      ColIcon.iconCache.set(cacheKey, cleanedSvg);
      return cleanedSvg;
    } catch (error) {
      console.error(`Error loading icon ${name}:`, error);
      return '';
    }
  }

  async updated(changedProperties: Map<string, unknown>) {
    if (changedProperties.has('name') || changedProperties.has('color')) {
      this.svgContent = await this.loadIcon(this.name);
    }
  }

  render() {
    // Ensure size is always valid, with a default fallback
    const sizeValue =
      typeof this.size === 'number'
        ? `${this.size}px`
        : typeof this.size === 'string' && this.size
          ? /^\d+$/.test(this.size)
            ? `${this.size}px`
            : this.size
          : '24px';

    return html`
      <style>
        :host {
          display: inline-block;
          width: ${sizeValue};
          height: ${sizeValue};
          line-height: 0;
        }
        .icon-container {
          width: 100%;
          height: 100%;
        }
        .icon-container > svg {
          display: block;
          width: 100%;
          height: 100%;
        }
        .icon-container > svg > path {
          display: block;
          width: 100%;
          height: 100%;
        }
      </style>
      <div class="icon-container">${this.svgContent ? unsafeHTML(this.svgContent) : ''}</div>
    `;
  }
}

import { Plugin, ResolvedConfig } from 'vite';
import { resolve } from 'path';
import * as fs from 'fs';
import { globSync } from 'glob';

/**
 * Configuration options for the Icons List Generator plugin
 */
export interface IconsListPluginOptions {
  /**
   * Source directory for icon files (relative to project root)
   * @default 'src/icons'
   */
  iconsDir?: string;

  /**
   * Output directory for generated files (relative to output directory)
   * @default 'icons-list'
   */
  outputDir?: string;

  /**
   * Whether to update package.json with exports entry
   * @default true
   */
  updatePackageJson?: boolean;

  /**
   * Export path to use in package.json
   * @default './icons-list'
   */
  exportPath?: string;
}

/**
 * Vite plugin for generating a list of available icons during build.
 * This plugin scans the icons directory and creates:
 * 1. A JavaScript file with an array of all icon names
 * 2. A TypeScript declaration file with proper typing
 * 3. Updates package.json to enable direct imports (optional)
 */
export default function iconsListPlugin(options: IconsListPluginOptions = {}): Plugin {
  const config: Required<IconsListPluginOptions> = {
    iconsDir: options.iconsDir || 'src/icons',
    outputDir: options.outputDir || 'icons-list',
    updatePackageJson: options.updatePackageJson !== false,
    exportPath: options.exportPath || './icons-list',
  };

  let resolvedConfig: ResolvedConfig;

  return {
    name: 'generate-icons-list',
    apply: 'build',

    configResolved(viteConfig) {
      resolvedConfig = viteConfig;
    },

    closeBundle() {
      const rootDir = resolvedConfig.root;
      const iconsDir = resolve(rootDir, config.iconsDir);

      if (!fs.existsSync(iconsDir)) {
        console.warn(`Icons directory '${iconsDir}' does not exist.`);
        return;
      }

      // Find all SVG files in the icons directory
      const svgFiles = globSync('**/*.svg', { cwd: iconsDir });

      // Extract icon names from filenames and sort them alphabetically
      const iconNames = svgFiles
        .map(file => {
          const filename = file.split('/').pop() || '';
          return filename.replace('.svg', '');
        })
        .sort();

      // Generate JavaScript content with the list of icons
      const iconsFileContent = `export const icons = ${JSON.stringify(iconNames, null, 2)};`;

      // Generate TypeScript declaration with proper typing
      const iconType = iconNames.map(name => `'${name}'`).join(' | ');
      const typesFileContent = `export declare const icons: Array<${iconType}>;\nexport type IconName = ${iconType};\n`;

      // Create output directory in dist
      const distDir = resolve(rootDir, resolvedConfig.build.outDir || 'dist');
      if (!fs.existsSync(distDir)) {
        fs.mkdirSync(distDir, { recursive: true });
      }

      // Create icons-list directory
      const iconsListDir = resolve(distDir, config.outputDir);
      if (!fs.existsSync(iconsListDir)) {
        fs.mkdirSync(iconsListDir, { recursive: true });
      }

      // Write files
      fs.writeFileSync(resolve(iconsListDir, 'index.js'), iconsFileContent);
      fs.writeFileSync(resolve(iconsListDir, 'index.d.ts'), typesFileContent);

      console.log(`Generated icons list with ${iconNames.length} icons`);

      // Update package.json if enabled
      if (config.updatePackageJson) {
        updatePackageJsonExports(rootDir, config.exportPath, config.outputDir);
      }
    },
  };
}

/**
 * Updates the package.json file to add the icons-list export path
 */
function updatePackageJsonExports(rootDir: string, exportPath: string, outputDir: string): void {
  const packageJsonPath = resolve(rootDir, 'package.json');

  if (fs.existsSync(packageJsonPath)) {
    try {
      // Read the original file content to preserve trailing newline
      const originalContent = fs.readFileSync(packageJsonPath, 'utf-8');
      const endsWithNewline = originalContent.endsWith('\n');

      const pkg = JSON.parse(originalContent);

      if (!pkg.exports) {
        pkg.exports = {};
      }

      pkg.exports[exportPath] = {
        import: `./dist/${outputDir}/index.js`,
        types: `./dist/${outputDir}/index.d.ts`,
      };

      // Stringify with pretty formatting (2 spaces)
      let newContent = JSON.stringify(pkg, null, 2);

      // Ensure we preserve the trailing newline if it existed in the original file
      if (endsWithNewline && !newContent.endsWith('\n')) {
        newContent += '\n';
      }

      fs.writeFileSync(packageJsonPath, newContent);
      console.log(`Updated package.json with exports entry for '${exportPath}'`);
    } catch (error) {
      console.error('Error updating package.json:', error);
    }
  } else {
    console.warn('package.json not found at', packageJsonPath);
  }
}

import { Plugin } from 'vite';
import * as fs from 'fs';

/**
 * Configuration options for the SVG Loader plugin
 */
export interface SvgLoaderPluginOptions {
  /**
   * Whether to minify SVG content by removing whitespace and newlines
   * @default true
   */
  minify?: boolean;

  /**
   * Custom transform function to apply to SVG content
   * @param svgContent The raw SVG content as a string
   * @returns The transformed SVG content
   */
  transform?: (svgContent: string) => string;
}

/**
 * Default SVG transformation that cleans and minifies SVG content
 */
function defaultSvgTransform(svgContent: string): string {
  return svgContent
    .trim()
    .replace(/>\s+</g, '><')
    .replace(/\s{2,}/g, ' ')
    .replace(/[\n\r]/g, '');
}

/**
 * Vite plugin for loading SVG files as strings.
 * This plugin preprocesses SVG files to optimize them and make them available as importable modules.
 */
export default function svgLoaderPlugin(options: SvgLoaderPluginOptions = {}): Plugin {
  const minify = options.minify !== false;
  const transform =
    options.transform || (minify ? defaultSvgTransform : (content: string) => content);

  return {
    name: 'svg-loader',
    enforce: 'pre',

    load(id) {
      if (id.endsWith('.svg')) {
        try {
          const svgContent = fs.readFileSync(id, 'utf-8');
          const transformedSvg = transform(svgContent);

          return `export default ${JSON.stringify(transformedSvg)}`;
        } catch (error) {
          console.error(`Error processing SVG file ${id}:`, error);
          return null;
        }
      }
      return null;
    },
  };
}
